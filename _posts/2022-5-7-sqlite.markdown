数据库操作
```
# 创建数据库
sqlite3 DatabaseName.db

# 查看数据库
sqlite>.databases

# 退出sqlite命令行
sqlite>.quit

# 导出数据库
$sqlite3 testDB.db .dump > testDB.sql

# 导入数据库
$sqlite3 testDB.db < testDB.sql
```

数据表操作
```
# 查看表
sqlite>.tables

# 查看表结构
sqlite>.schema COMPANY

# 创建表
sqlite> CREATE TABLE COMPANY(
   ID INT PRIMARY KEY     NOT NULL,
   NAME           TEXT    NOT NULL,
   AGE            INT     NOT NULL,
   ADDRESS        CHAR(50),
   SALARY         REAL
);
```
